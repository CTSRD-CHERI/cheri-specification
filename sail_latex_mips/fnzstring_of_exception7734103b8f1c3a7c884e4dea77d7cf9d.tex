function #\hyperref[sailMIPSzstringzyofzyexception]{string\_of\_exception}#(ex) : Exception -> string =
   match ex {
      Interrupt    => "Interrupt",
      TLBMod       => "TLBMod",
      TLBL         => "TLBL",
      TLBS         => "TLBS",
      AdEL         => "AdEL",
      AdES         => "AdES",
      Sys          => "Sys",
      Bp           => "Bp  ",
      ResI         => "ResI",
      CpU          => "CpU",
      Ov           => "Ov",
      Tr           => "Tr",
      C2E          => "C2E",
      C2Trap       => "C2Trap",
      XTLBRefillL  => "XTLBRefillL",
      XTLBRefillS  => "XTLBRefillS",
      XTLBInvL     => "XTLBInvL",
      XTLBInvS     => "XTLBInvS",
      MCheck       => "MCheck"
   }
