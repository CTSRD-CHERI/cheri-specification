function #\hyperref[sailMIPSzunalignedBytesTouched]{unalignedBytesTouched}#(vAddr : int, width : WordTypeUnaligned) -> (int, int) = {
  woffset = vAddr % 4;
  doffset = vAddr % 8;
  match width {
    WL => (vAddr, 4 - woffset),
    WR => (vAddr - woffset, woffset + 1),
    DL => (vAddr, 8 - doffset),
    DR => (vAddr - doffset, doffset + 1)
  }
}
